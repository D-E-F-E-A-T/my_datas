数据加载及模板显示

收藏数据加载

  • 对用户收藏的新闻进行分页返回，默认返回第1页
    • 返回的数据中要告诉前端 总页数、当前页数
@profile_blu.route('/collection')
@login_required
def user_collection_news():
    """
    用户收藏新闻信息:
    """
    # 1. 接收参数并进行参数校验
    page = request.args.get('p', 1)

    try:
        page = int(page)
    except Exception as e:
        current_app.logger.error(e)
        return jsonify(errno=RET.PARAMERR, errmsg='参数错误')

    # 2. 获取用户收藏的新闻信息并进行分页
    # 获取登录用户
    user = g.user
    try:
        pagination = user.collection_news.paginate(page, constants.USER_COLLECTION_MAX_NEWS, False)
        collection_news = pagination.items
        total_page = pagination.pages
        current_page = pagination.page
    except Exception as e:
        current_app.logger.error(e)
        return jsonify(errno=RET.DBERR, errmsg='查询用户收藏新闻信息失败')

    # 3. 使用模板
    return render_template('news/user_collection.html',
                           collection_news=collection_news,
                           total_page=total_page,
                           current_page=current_page)

前端模板展示

<div class="my_collect">
    <h3>我的收藏</h3>
    <ul class="article_list">
        {# 遍历显示`用户收藏`新闻信息 #}
        {% for news in collection_news %}
        <li><a href="#">{{ news.title }}</a><span>{{ news.create_time }}</span></li>
        {% endfor %}
    </ul>


    <div id="pagination" class="page"></div>
    <script>
        $(function() {
            $("#pagination").pagination({
                currentPage: {{ current_page }},
                totalPage: {{ total_page }},
                callback: function(current) {
                    // alert('ok!');
                    location.href = '/user/collection?p=' + current;
                }
            });
        });
    </script>
</div>
