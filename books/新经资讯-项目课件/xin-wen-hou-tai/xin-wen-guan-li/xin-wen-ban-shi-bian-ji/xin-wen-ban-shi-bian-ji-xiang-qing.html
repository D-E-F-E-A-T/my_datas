                           news=news,
                           categories=categories)

新闻编辑提交

  • 修改视图函数，使其能够接受POST请求执行保存编辑操作，并实现其代码
@admin_blu.route('/news/edit/<int:news_id>', methods=['GET', 'POST'])
@admin_login_required
def news_edit_detail(news_id):
    """
    后台管理新闻详情编辑页面:
    """
    if request.method == 'GET':
        # 1. 根据`news_id`查询新闻信息
        try:
            news = News.query.get(news_id)
        except Exception as e:
            current_app.logger.error(e)
            abort(500)

        if not news:
            # 新闻信息不存在
            abort(404)

        # 2. 获取`新闻分类`信息
        try:
            categories = Category.query.all()
        except Exception as e:
            current_app.logger.error(e)
            abort(500)

        # 去除`最新`分类
        categories.pop(0)

        # 3. 使用模板文件
        return render_template('admin/news_edit_detail.html',
                               news=news,
                               categories=categories)
    else:
        # 执行编辑处理
        # 1. 接收参数并进行参数校验
        title = request.form.get("title")
        digest = request.form.get("digest")
        content = request.form.get("content")
        index_image = request.files.get("index_image")
        category_id = request.form.get("category_id")

        if not all([title, digest, content, category_id]):
            return jsonify(errno=RET.PARAMERR, errmsg='参数不完整')

        try:
            category_id = int(category_id)
        except Exception as e:
            current_app.logger.error(e)
            return jsonify(errno=RET.PARAMERR, errmsg='参数错误')

        # 2. 根据`news_id`查询新闻信息(如果查不到，说明新闻信息不存在)
         try:
             news = News.query.get(news_id)
         except Exception as e:
             current_app.logger.error(e)
             return jsonify(errno=RET.DBERR, errmsg='查询新闻信息失败')

         if not news:
             return jsonify(errno=RET.NODATA, errmsg='新闻信息不存在')

        # 3. 将索引图片上传至七牛云
        if index_image:
            try:
                key = storage(index_image.read())
            except Exception as e:
                current_app.logger.error(e)
                return jsonify(errno=RET.THIRDERR, errmsg='上传文件到七牛云失败')

        # 4. 设置`新闻的信息`
        news.title = title
        news.digest = digest
        news.content = content
        news.category_id = category_id
        if index_image:
            news.index_image_url = constants.QINIU_DOMIN_PREFIX + key

        # 5. 将修改信息添加进数据库
        try:
            db.session.commit()
        except Exception as e:
            db.session.rollback()
            current_app.logger.error(e)
            return jsonify(errno=RET.DBERR, errmsg='保存新闻信息失败')

        # 6. 返回应答，编辑成功
        return jsonify(errno=RET.OK, errmsg='编辑成功')
  • 在 static/admin/js/news_edit_detail.js 中添加以下代码
$(function(){
    $(".news_edit").submit(function (e) {
        // 阻止表单默认提交
        e.preventDefault();

        // 获取新闻的id
        var news_id = $(".news_edit").attr("data-news-id");

        // TODO 新闻编辑提交
        $(this).ajaxSubmit({
            beforeSubmit: function (request) {
                // 在提交之前，对参数进行处理
                for(var i=0; i<request.length; i++) {
                    var item = request[i];
                    if (item["name"] == "content") {
                        item["value"] = tinyMCE.activeEditor.getContent()
                    }
                }
            },
            url: "/admin/news/edit/" + news_id,
            type: "POST",
            headers: {
                "X-CSRFToken": getCookie('csrf_token')
            },
            success: function (resp) {
                if (resp.errno == "0") {
                    // `编辑新闻`成功
                    alert('编辑成功');
                }
                else {
                    // `编辑新闻`失败
                    alert(resp.errmsg);
                }
            }
        })
    })
});
